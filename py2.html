def knn_1(x,y,z,k):
    x=np.array(x); y=np.array(y)
    dis=[]
    for i in range(x.shape[0]):
        dis.append(np.linalg.norm(z-x[i,]))
    S=np.argsort(dis)[0:k]  #kindex
    u=np.bincount(y[S])     #
    m = [i for i, x in enumerate(u) if x == max(u)] #index
    #()
    while (len(m)>1):
        k=k-1
        S=S[0:k]
        u=np.bincount(y[S])
        m = [i for i, x in enumerate(u) if x == max(u)] #index
    return m[0]

def cv_linear(X,y,K):
    n=len(y);m=int(n/K)
    S=0
    for j in range(K):
        test=list(range(j*m,(j+1)*m)) #  
        train=list(set(range(n))-set(test))     # 
        beta=np.linalg.inv(X[train,].T@X[train,])@X[train,].T@y[train]
        e=y[test]-X[test,]@beta
        S=S+np.linalg.norm(e)**2
    return S/n

def knn_1(x,y,z,k):   # 
    x=np.array(x)
    n,p=x.shape
    dis=np.zeros(n)
    for i in range(n):
        dis[i]=(z-x[i,]).T@(z-x[i,])
    S=np.argsort(dis)[0:k]  # kindex
    u=np.bincount(y[S])     # 
    m = [i for i, x in enumerate(u) if x == max(u)] # index
    # ()
    while (len(m)>1):
        k=k-1
        S=S[0:k]
        u=np.bincount(y[S])
        m = [i for i, x in enumerate(u) if x == max(u)] # index
    return m[0]
def knn(x,y,z,k):
    n=z.shape[0]
    w=np.zeros(n)
    for i in range(n):
        w[i]=knn_1(x,y,z[i,],k)
    return w

def cv_fast(X,y,k):
    n=len(y)
    m=n/k
    H=X@np.linalg.inv(X.T@X)@X.T
    I=np.diag(np.repeat(1,n))
    e=(I-H)@y
    I=np.diag(np.repeat(1,m))
    S=0
    for j in range(k):
        test=np.arange(j*m,(j+1)*m,1,dtype=int)
        S=S+(np.linalg.inv(I-H[test,test])@e[test]).T@np.linalg.inv(I-H[test,test])@e[test]
    return S/n

def bt(df,f,r):
    m=df.shape[0]
    org=f(df,np.arange(0,m,1))
    u=[]
    for j in range(r):
        index=np.random.choice(m, m, replace=True) 
        u.append(f(df,index))
    return {'original':org, 'bias':np.mean(u)-org, 'stderr':np.std(u)}
